#define RELAY 4
#define SENSOR 7
#define ZERO_RAW 1850  // CHANGE THIS to your actual zero reading
#define SENSITIVITY 160.0

void setup() {
  Serial.begin(9600);
  pinMode(RELAY, OUTPUT);
  digitalWrite(RELAY, LOW);
}

void loop() {
  int raw = analogRead(SENSOR);
  float current = (raw - ZERO_RAW) / SENSITIVITY;
  
  Serial.print("Raw: ");
  Serial.print(raw);
  Serial.print(" | Current: ");
  Serial.print(current, 3);
  Serial.println("A");
  
  delay(1000);
  
  static byte count = 0;
  count++;
  
  if (count == 10) {
    digitalWrite(RELAY, HIGH);
    delay(5000);
    digitalWrite(RELAY, LOW);
    count = 0;
  }
}
